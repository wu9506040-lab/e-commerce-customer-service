```python
class RefundReasonAdvice(BaseModel):
    """退款原因填写指导"""
    order_no: str
    reason_category: str
    suggested_reason_text: str          # 建议填写的具体文字
    success_rate_hint: str             # "高" / "中" / "低"（基于 reason_category）
    evidence_required: List[str]        # 需要的凭证（照片/视频/聊天截图）
    additional_tips: List[str]          # 额外提示（如"不要写'不想要了'"）


class ShippingInsuranceInfo(BaseModel):
    """运费险规则"""
    order_no: str
    insured: bool                       # 该订单是否购买运费险
    coverage_amount: Optional[float]    # 赔付额度（最高 X 元）
    eligible: bool                      # 当前情况是否符合理赔条件
    estimated_payout_days: Optional[int]  # 预计到账天数
    notes: List[str]                    # 注意事项


class RefundTypeAdvice(BaseModel):
    """仅退款 vs 退货退款建议"""
    order_no: str
    recommended_type: str               # "refund_only" / "return_and_refund"
    reasoning: str                      # 推荐理由（中文）
    conditions: List[str]               # 适用条件列表
```

### 3.2 Schema

```python
class Promotion(BaseModel):
    """优惠活动"""
    promotion_id: str
    name: str                          # "双 11 满 300 减 50"
    type: str                          # "full_reduction" / "discount" / "gift"
    threshold: Optional[float] = None  # 满 X 触发
    benefit: Optional[float] = None    # 减 Y
    applicable_stores: List[str]       # 适用店铺（空 = 全店）
    applicable_categories: List[str]   # 适用类目（空 = 全部）
    start_time: datetime
    end_time: datetime
    stackable: bool                    # 能否与其他优惠叠加


class CouponStackResult(BaseModel):
    """优惠券叠加校验结果"""
    stackable_groups: List[List[str]]  # 可叠加的 coupon_id 分组
    conflicting_pairs: List[dict]      # 互斥的券对：[{a, b, reason}]
    best_combination: List[str]        # 推荐的券组合（优惠最大化）


class BundleDiscountResult(BaseModel):
    """跨店满减计算结果"""
    current_total: float               # 当前所有店合计
    store_totals: Dict[str, float]     # 每店金额
    next_threshold: Optional[float]    # 距离下一档满减还差多少
    next_benefit: Optional[float]      # 下一档满减可减多少
    suggestion: Optional[str]          # 凑单建议："A 店再买 50 元可减 30"
```

```python
    """

    async def modify_address(

    ) -> ModifyResult:
        """修改收货地址（限未发货订单）"""
        ...

    async def modify_item_spec(
        self, user_id: int, order_no: str, sku: str, new_qty: Optional[int] = None,
    ) -> ModifyResult:
        """修改商品规格/数量（限未发货订单）
        
        注意：换 SKU 需谨慎（V2 仅支持数量调整；换 SKU 留 V3+）
        """
        ...

    async def merge_orders(
        self, user_id: int, order_nos: List[str],
    ) -> MergeResult:
        """合并订单（限同一店铺 + 未发货 + 5 分钟内）"""
        ...
```

### 4.2 Schema

```python
class ModifyResult(BaseModel):
    """修改结果"""
    success: bool
    order_no: str
    modification_type: str             # "address" / "spec" / "merge"
    reason: str                        # 中文提示（成功/失败原因）
    before_snapshot: Optional[dict]    # 修改前快照（便于回滚）
    after_snapshot: Optional[dict]     # 修改后快照
```

### 4.3 默认实现（`app/services/order_modify/mysql_impl.py`）

```python
class MySQLOrderModifyService:
    async def modify_address(self, user_id, order_no, new_address):
        # 1. 查 OrderService.get_order(user_id, order_no) 验证越权 + 拿 status
        # 2. 检查 status in ("pending", "paid") 才允许修改
        # 3. UPDATE orders SET shipping_address = ?
        # 4. 返 ModifyResult

    async def modify_item_spec(self, user_id, order_no, sku, new_qty):
        # 1. 验证越权 + status
        # 2. 查 order_items WHERE order_id = ? AND sku = ?
        # 3. UPDATE qty + subtotal
        # 4. 返 ModifyResult

    async def merge_orders(self, user_id, order_nos):
        # 1. 查所有订单，验证同 user + 同 store + 都未发货 + 5 分钟内
        # 2. 选择最早订单为主订单
        # 3. 把其他订单的 order_items 合并到主订单
        # 4. UPDATE 状态为"merged"或删除
        # 5. 返 MergeResult
```

**修改限制规则（硬编码或 YAML）**：
- `pending` / `paid`：允许修改
- `shipped` / `delivered` / `completed` / `refunded`：禁止修改

### 4.4 异常类

```python
class ModifyError(Exception): ...
class ModifyNotAllowedError(ModifyError): ...   # 状态不允许修改
class OrderNotFoundError(ModifyError): ...      # 订单不存在或越权
class MergeConditionError(ModifyError): ...     # 合并条件不满足
```
